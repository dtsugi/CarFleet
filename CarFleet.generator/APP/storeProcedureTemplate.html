USE ;
DROP PROCEDURE IF EXISTS ;
DELIMITER $$
CREATE PROCEDURE 
(
  
#divFields
 
)
BEGIN

/* INSERT */
IF OPERACION_OUT = 0 THEN 

#divInsert
 
/* DELETE */
ELSEIF OPERACION_OUT = 1 THEN 

#divDelete
  
/* UPDATE */
ELSEIF OPERACION_OUT = 2 THEN 

#divUpdate
  
/* SELECT BY ID */
ELSEIF OPERACION_OUT = 3 THEN 

#divSelectById

/* SELECT ALL */
ELSEIF OPERACION_OUT = 4 THEN 

#divSelectAll

END IF;

END
